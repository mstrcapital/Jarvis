## 🛠️ 工具支持

### 收入追踪系统

```python
# Jarvis营收追踪
class JarvisRevenue:
    def __init__(self):
        self.monthly_target = 100000
        self.current_month = None
        self.revenue = {}
        self.expenses = {}
    
    def record_income(self, source, amount):
        if source not in self.revenue:
            self.revenue[source] = []
        self.revenue[source].append({
            "amount": amount,
            "date": datetime.now()
        })
    
    def get_status(self):
        total = sum(sum(r) for r in self.revenue.values())
        return {
            "total": total,
            "target": self.monthly_target,
            "progress": total / self.monthly_target,
            "status": "healthy" if total > self.monthly_target * 0.5 else "warning"
        }
```

### 任务管理系统

```python
class JarvisTasks:
    def __init__(self):
        self.tasks = []
    
    def add_task(self, task, deadline, owner):
        self.tasks.append({
            "task": task,
            "deadline": deadline,
            "owner": owner,
            "status": "pending"
        })
    
    def daily_check(self):
        overdue = [t for t in self.tasks if t["status"] == "overdue"]
        return overdue
```

### 客户关系管理

```python
class JarvisCRM:
    def __init__(self):
        self.customers = []
    
    def add_lead(self, source, interest, budget):
        self.customers.append({
            "source": source,
            "interest": interest,
            "budget": budget,
            "status": "lead",
            "created": datetime.now()
        })
    
    def get_pipeline(self):
        stages = {"lead": 0, "qualified": 0, "proposal": 0, "won": 0, "lost": 0}
        for c in self.customers:
            stages[c["status"]] += 1
        return stages
```
